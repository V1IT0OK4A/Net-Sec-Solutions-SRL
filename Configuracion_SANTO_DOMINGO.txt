enable
conf t
hostname SAN-DOM

ip cef

username admin privilege 15 secret N3tS3c-2025
ip domain-name netsec-solutions.com.do
crypto key generate rsa modulus 1024
aaa new-model
radius-server host 10.17.0.43 auth-port 1812 acct-port 1813 key NetSecSecret123
aaa authentication login default group radius local
aaa authorization exec default group radius local 
ip ssh version 2

access-list 7 remark SOLO-ACCESO-TI-STGO
access-list 7 permit 10.17.0.0 0.0.0.31
access-list 7 deny any 

line vty 0 4
 transport input ssh
 login authentication default
 access-class 7 in
exit

crypto isakmp nat-keepalive 20

interface e0/2
 description CONECTION TO ISP
 ip address dhcp
 ip nat outside
 duplex full
 no shutdown
exit

interface e0/0
 description CONECTION TO SWM1
 ip address 10.16.2.81 255.255.255.252
 ip ospf network point-to-point
 ip nat inside
 duplex full
 no shutdown
exit

interface e0/1
 description CONECTION TO SWM2
 ip address 10.16.2.85 255.255.255.252
 ip ospf network point-to-point
 ip nat inside
 duplex full
 no shutdown
exit

crypto isakmp policy 1
 encryption aes
 hash sha512
 authentication pre-share
 group 14
 lifetime 86400
exit

crypto isakmp key N3tS3cVPN address 0.0.0.0 0.0.0.0
crypto isakmp invalid-spi-recovery

crypto ipsec transform-set NS-VPN esp-aes esp-sha512-hmac
 mode transport
exit

crypto ipsec profile NETSEC-VPN
 set transform-set NS-VPN
exit

interface Tunnel1
 description *** DMVPN SPOKE TO SANTIAGO ***
 ip address 10.16.2.73 255.255.255.248 
 tunnel source e0/2
 tunnel mode gre multipoint
 tunnel key 10
 tunnel protection ipsec profile NETSEC-VPN
 
 ip nhrp network-id 10
 ip nhrp authentication NetSec
 ip nhrp map multicast dynamic
 ip nhrp map 10.16.2.74 1.0.0.9
 ip nhrp map multicast 1.0.0.9
 ip nhrp nhs 10.16.2.74
 
 ip mtu 1400
 ip tcp adjust-mss 1360
 ip ospf network point-to-multipoint
 ip ospf hello-interval 30
 ip ospf mtu-ignore
 ip ospf priority 0
 no shutdown
exit

access-list 1 permit 10.16.0.0 0.0.255.255
ip nat inside source list 1 interface e0/2 overload

ip route 0.0.0.0 0.0.0.0 Ethernet0/2

router ospf 1
 router-id 2.2.2.2
 network 10.16.2.80 0.0.0.3 area 0
 network 10.16.2.84 0.0.0.3 area 0
 network 10.16.2.72 0.0.0.7 area 0
 default-information originate always
exit
